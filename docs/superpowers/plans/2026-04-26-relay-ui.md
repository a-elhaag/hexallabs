# The Relay UI — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Wire the existing relay backend to a full UI — an ordered chain selector (2–7 models) in the SlashMenu, a `RelayChain` visualization component with animated baton-pass between model cards, and the backend extended to accept chains longer than 2 by looping the existing relay stages.

**Architecture:** The backend `_relay_stream` currently hard-codes a two-stage A→B loop; extend it into an N-stage loop where each completed stage's output becomes the next stage's baton, emitting `relay_handoff` between each pair. The frontend adds an ordered-list relay chain selector in `SlashMenu`, replaces the raw `ChatMessage` append pattern in `ChatWindow` with a structured `relay` Turn type, and renders a new `RelayChain` component that animates handoffs via CSS transitions.

**Tech Stack:** FastAPI async Python, Next.js App Router TypeScript strict, Tailwind, SSE via fetch+ReadableStream, React state + refs, no new npm packages.

def _parse_sse(body: str) -> list[tuple[str, dict[str, object]]]:
    events: list[tuple[str, dict[str, object]]] = []
    for block in body.split("\n\n"):
        block = block.strip("\n")
        if not block or block.startswith(":"): continue
        event, data = None, None
        for line in block.split("\n"):
            if line.startswith("event: "): event = line[7:]
            elif line.startswith("data: "): data = line[6:]
        if event and data is not None:
            events.append((event, json.loads(data)))
    return events

```tsx
interface RelayChainSelectorProps {
  chain: ModelName[]
  onChange: (chain: ModelName[]) => void
}

function RelayChainSelector({ chain, onChange }: RelayChainSelectorProps) {
  const available = MODELS.filter(m => !chain.includes(m))
  const canAdd = chain.length < 7
  const canRemove = chain.length > 2

  function move(idx: number, dir: -1 | 1) {
    const next = [...chain]
    const target = idx + dir
    if (target < 0 || target >= next.length) return
    ;[next[idx], next[target]] = [next[target], next[idx]]
    onChange(next)
  }

  function remove(idx: number) {
    if (!canRemove) return
    onChange(chain.filter((_, i) => i !== idx))
  }

  function add(m: ModelName) {
    if (!canAdd) return
    onChange([...chain, m])
  }

  return (
    <div className="flex flex-col gap-1.5">
      <p className="text-[10px] font-bold text-[#a89080] uppercase tracking-widest px-1">
        Chain — {chain.length} model{chain.length !== 1 ? 's' : ''}
      </p>
      {chain.map((m, idx) => (
        <div key={m} className="flex items-center gap-2 px-2 py-1.5 bg-[#6290c3]/8 rounded-lg" style={{ transition: 'all 300ms cubic-bezier(0.16,1,0.3,1)' }}>
          <span className="text-[10px] font-black text-[#6290c3]/60 w-4 text-center">{idx + 1}</span>
          <span className="text-sm font-semibold text-[#6290c3] flex-1">{MODEL_DISPLAY[m]}</span>
          <div className="flex gap-0.5">
            <button onClick={() => move(idx, -1)} disabled={idx === 0} className="w-5 h-5 flex items-center justify-center text-[#a89080] hover:text-[#2c2c2c] disabled:opacity-20 rounded transition-colors" aria-label="Move up">↑</button>
            <button onClick={() => move(idx, 1)} disabled={idx === chain.length - 1} className="w-5 h-5 flex items-center justify-center text-[#a89080] hover:text-[#2c2c2c] disabled:opacity-20 rounded transition-colors" aria-label="Move down">↓</button>
            <button onClick={() => remove(idx)} disabled={!canRemove} className="w-5 h-5 flex items-center justify-center text-[#a89080] hover:text-red-500 disabled:opacity-20 rounded transition-colors text-xs" aria-label="Remove">✕</button>
          </div>
        </div>
      ))}
      {canAdd && available.length > 0 && (
        <div className="mt-0.5">
          <p className="text-[10px] font-bold text-[#a89080]/60 uppercase tracking-widest px-1 mb-1">Add to chain</p>
          {available.map(m => (
            <button key={m} onClick={() => add(m)} className="flex items-center gap-2 w-full px-2 py-1.5 rounded-lg text-sm text-[#2c2c2c] hover:bg-[#2c2c2c]/5 transition-colors">
              <span className="w-3 h-3 rounded-full border-2 border-[#a89080]/40 flex-shrink-0" />
              <span className="font-medium">{MODEL_DISPLAY[m]}</span>
            </button>
          ))}
        </div>
      )}
      {!canAdd && <p className="text-[10px] text-[#a89080]/50 px-1">Maximum 7 models reached</p>}
    </div>
  )
}
```

'use client'
import { useEffect, useRef } from 'react'
import ReactMarkdown from 'react-markdown'
import { MODEL_DISPLAY, ModelName, RelayStage, RelayHandoffState } from '@/lib/types'

function StreamingCursor() {
  return <span className="inline-block w-0.75 h-[1em] bg-current rounded-full ml-0.5 align-middle" style={{ animation: 'cursorBlink 1s ease-in-out infinite' }} />
}

function BatonConnector({ handoff }: { handoff: RelayHandoffState | undefined }) {
  const isDone = !!handoff && !handoff.animating
  const isAnimating = !!handoff?.animating
  return (
    <div className="flex items-center justify-center w-10 shrink-0 relative">
      <div className="h-0.5 w-full transition-colors duration-300" style={{ backgroundColor: isDone || isAnimating ? '#6290c3' : '#a89080' }} />
      <div className="absolute right-0 transition-colors duration-300" style={{ width: 0, height: 0, borderTop: '4px solid transparent', borderBottom: '4px solid transparent', borderLeft: `6px solid ${isDone || isAnimating ? '#6290c3' : '#a89080'}` }} />
      {isAnimating && (
        <div className="absolute top-1/2 -translate-y-1/2 w-2.5 h-2.5 rounded-full bg-[#6290c3] shadow-sm" style={{ left: 0, animation: 'batonTravel 0.55s cubic-bezier(0.16,1,0.3,1) forwards' }} />
      )}
    </div>
  )
}

function RelayCard({ stage, isActive }: { stage: RelayStage; isActive: boolean }) {
  const ref = useRef<HTMLDivElement>(null)
  useEffect(() => {
    const el = ref.current
    if (!el) return
    el.style.animation = 'msgIn 0.28s cubic-bezier(0.16,1,0.3,1) both'
  }, [])
  const hasContent = stage.content.length > 0
  return (
    <div ref={ref} className="flex flex-col rounded-2xl border p-4 min-w-0 flex-1 max-w-xs transition-all duration-300"
      style={{
        backgroundColor: '#f5f1ed',
        borderColor: isActive ? '#6290c3' : stage.isDone ? '#6290c380' : '#a8908033',
        boxShadow: isActive ? '0 0 0 2px #6290c340, 0 4px 12px #6290c320' : stage.isDone ? '0 2px 6px #0000000a' : 'none',
        opacity: hasContent || isActive ? 1 : 0.45,
      }}>
      <div className="flex items-center justify-between mb-2 shrink-0">
        <p className="text-[10px] font-black text-[#a89080]/70 uppercase tracking-widest">{MODEL_DISPLAY[stage.model as ModelName] ?? stage.model}</p>
        {isActive && <span className="w-2 h-2 rounded-full bg-[#6290c3] shrink-0" style={{ animation: 'pulse 1.4s ease-in-out infinite' }} />}
        {stage.isDone && !isActive && <span className="text-[10px] text-[#6290c3] font-bold">✓</span>}
      </div>
      <div className="text-sm leading-relaxed text-[#2c2c2c] overflow-y-auto flex-1 min-h-0 max-h-52">
        {hasContent ? (
          <ReactMarkdown components={{
            p: ({ children }) => <p className="mb-2 last:mb-0">{children}</p>,
            code: ({ children, className }) => className ? <code className="text-xs font-mono">{children}</code> : <code className="bg-[#2c2c2c]/8 px-1 py-0.5 rounded-md text-xs font-mono">{children}</code>,
            pre: ({ children }) => <pre className="bg-[#2c2c2c]/8 rounded-xl p-3 overflow-x-auto text-xs font-mono my-2">{children}</pre>,
            ul: ({ children }) => <ul className="list-disc pl-4 mb-2 space-y-0.5">{children}</ul>,
            li: ({ children }) => <li>{children}</li>,
          }}>{stage.content}</ReactMarkdown>
        ) : (
          <span className="text-[#a89080]/50 text-xs">{isActive ? 'Generating…' : 'Waiting…'}</span>
        )}
        {stage.isStreaming && <StreamingCursor />}
      </div>
    </div>
  )
}

interface Props {
  stages: RelayStage[]
  handoffs: RelayHandoffState[]
}

export function RelayChain({ stages, handoffs }: Props) {
  if (stages.length === 0) return null
  return (
    <div className="w-full max-w-5xl mx-auto px-6 py-1.5">
      <div className="flex items-stretch gap-0 overflow-x-auto pb-1">
        {stages.map((stage, idx) => (
          <div key={stage.id} className="flex items-center min-w-0">
            <RelayCard stage={stage} isActive={stage.isStreaming} />
            {idx < stages.length - 1 && (
              <BatonConnector handoff={handoffs.find(h => h.from === stage.model)} />
            )}
          </div>
        ))}
      </div>
    </div>
  )
}
